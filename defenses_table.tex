% ---------------------------------------------------------------
% Total losses, per family
% ---------------------------------------------------------------
\paragraph{Circuit-breaker cells.}
The per-step training loss is
\begin{equation}
  \mathcal{L}_\text{CB}
  \;=\;
  w_\text{benign}\,\mathcal{L}^\text{CE}_\text{benign}
  \;+\;
  w_\text{kl}\,\mathcal{L}^\text{KL}_\text{benign}
  \;+\;
  w_\text{harm}\,\mathcal{L}_\text{CB-reg}
  \;+\;
  w_\text{pra}\,\mathcal{L}_\text{PRA},
  \label{eq:cb-loss}
\end{equation}
with weights from Table~\ref{tab:cb-loss}.

\paragraph{Triplet cells.}
The per-step training loss is
\begin{equation}
  \mathcal{L}_\text{tri}
  \;=\;
  \alpha\,\mathcal{L}_\text{safe}
  \;+\;
  \beta\,\mathcal{L}_\text{unsafe}
  \;+\;
  \gamma\,\mathcal{L}^\text{KL}_\text{benign}
  \;+\;
  w_\text{pra}\,\mathcal{L}_\text{PRA},
  \label{eq:triplet-loss}
\end{equation}
with weights from Table~\ref{tab:triplet-loss}.  Note that
Eq.~\ref{eq:triplet-loss} contains no cross-entropy term on either
harmful or benign target tokens; capability is preserved purely
through the KL retain term \(\mathcal{L}^\text{KL}_\text{benign}\).

% ---------------------------------------------------------------
% Loss components --- CE, KL, PRA
% ---------------------------------------------------------------
\paragraph{Benign cross-entropy.}
Standard supervised next-token cross-entropy on benign \emph{(prompt,
helpful-response)} pairs from the union of UltraChat and the
WildGuardMix-aligned compliance set.  With prompt tokens masked from
the loss, the only positions scored are the assistant tokens
\(t \in \mathcal{A}\).  For a sequence \(y_{1:T}\),
\begin{equation}
  \mathcal{L}^\text{CE}_\text{benign}
  \;=\;
  -\frac{1}{|\mathcal{A}|}
   \sum_{t \in \mathcal{A}}
   \log p_\theta\!\bigl(y_t \,\bigm|\, y_{<t}\bigr).
  \label{eq:benign-ce}
\end{equation}
This term is the standard SFT objective and trains the defended
model to keep producing the helpful response on benign inputs.  It
appears only in the CB total loss; the triplet family relies on the
KL term alone to preserve capability.

\paragraph{Benign KL retain.}
Top-\(k\) KL divergence between the defended-model and frozen base-model
distributions, evaluated on benign inputs.  Let \(p_\theta\) be the
distribution under the defended model (LoRA active) and \(p_\text{base}\)
the distribution under the same backbone with the LoRA adapter disabled:
\begin{equation}
  \mathcal{L}^\text{KL}_\text{benign}
  \;=\;
  \mathrm{KL}\!\bigl(\,p_\theta(\cdot \mid x) \;\big\|\; p_\text{base}(\cdot \mid x)\,\bigr),
  \quad x \sim \text{benign}.
  \label{eq:benign-kl}
\end{equation}
Unlike the CE term, which scores only the helpful target token, the
KL term anchors the entire output distribution, including refusals,
formatting, and stylistic choices.  It is present in both the CB and
triplet total losses.

\paragraph{Predictive Representation Alignment (PRA).}
\(\mathcal{L}_\text{PRA}\) is the auxiliary loss that distinguishes
the PRA cells from their matched no-PRA controls.  For each
\((\text{vanilla}, \text{adversarial})\) sibling pair drawn from the
WildJailbreak pool, both prompts are forwarded through the defended
encoder; the assistant-token mean-pooled hidden state at the
alignment layer is taken for each.  A small MLP predictor
\(g_\phi\) projects the adversarial representation
\(\mathbf{h}^{(\text{adv})}\) into the vanilla manifold, and the loss
is mean-squared error against the corresponding vanilla
representation \(\mathbf{h}^{(\text{van})}\) with stop-gradient on
the target:
\begin{equation}
  \mathcal{L}_\text{PRA}
  \;=\;
  \bigl\lVert\, g_\phi(\mathbf{h}^{(\text{adv})}) \;-\; \mathrm{sg}\bigl[\mathbf{h}^{(\text{van})}\bigr] \,\bigr\rVert_2^2.
  \label{eq:pra}
\end{equation}
Setting \(w_\text{pra}\!=\!0\) gives the matched-pair no-PRA control
in which every other term is identical, isolating the contribution
of representation alignment.

% ---------------------------------------------------------------
% Harm regularizers
% ---------------------------------------------------------------
\paragraph{Circuit-breaker regularizer.}
Following \citet{zou2024improving}.  Let
\(\mathbf{h}^{(\text{def})}_b,\mathbf{h}^{(\text{def})}_h\) be the
defended-model assistant-token mean-pooled hidden states at the
alignment layer for benign and harmful inputs, and
\(\mathbf{h}^{(\text{base})}_b,\mathbf{h}^{(\text{base})}_h\) the
corresponding states with the LoRA adapter disabled.  The CB
regularizer combines a benign anchor with a harmful break-away:
\begin{equation}
  \mathcal{L}_\text{CB-reg}
  \;=\;
  \alpha_\text{cb}\,\bigl\lVert \mathbf{h}^{(\text{def})}_b - \mathbf{h}^{(\text{base})}_b \bigr\rVert_2^2
  \;+\;
  \beta_\text{cb}(t)\,\operatorname{ReLU}\!\Bigl(\cos\!\bigl(\mathbf{h}^{(\text{def})}_h,\,\mathbf{h}^{(\text{base})}_h\bigr)\Bigr).
  \label{eq:cb-reg}
\end{equation}
The first term anchors benign representations to the base.  The
second pushes harmful representations to be orthogonal to their base
counterpart; the ReLU clips the loss once cosine drops to zero, so
the loss does not pull representations past \(90^\circ\).
\(\beta_\text{cb}(t)\) is a cosine schedule starting at
\(10\beta_\text{cb}\) and decaying to \(\beta_\text{cb}\) over the
full run, encouraging early aggressive disruption.

\paragraph{Triplet regularizer.}
Proposed by \citet{simko2025}.  Two margin-hinge triplets in
representation space, one per polarity.  Let
\(\mathcal{S}=\{20,21,\dots,30\}\) be the layer stack used for
representations, and let \(\mathbf{c}_h = \tfrac{1}{n_h}\sum_i
\mathbf{h}^{(\text{def})}_{h,i}\) be the in-batch harmful centroid.
The hybrid distance combines \(\ell_2\) with a clipped-cosine penalty,
\begin{equation}
  d(\mathbf{x},\mathbf{y})
  \;=\;
  \lVert \mathbf{x}-\mathbf{y}\rVert_2
  \;+\;
  10\,\operatorname{ReLU}\!\bigl(1 - \cos(\bar{\mathbf{x}},\bar{\mathbf{y}})\bigr),
  \label{eq:triplet-d}
\end{equation}
where \(\bar{\mathbf{x}}\) is the \(\ell_2\)-normalized
\(\mathbf{x}\).  The two hinges are
\begin{align}
  \mathcal{L}_\text{safe}
  &= \frac{1}{n_b}\sum_{j=1}^{n_b}
     \operatorname{ReLU}\!\Bigl(
       d\!\bigl(\mathbf{h}^{(\text{def})}_{b,j},\,\mathbf{h}^{(\text{base})}_{b,j}\bigr)
       - d\!\bigl(\mathbf{h}^{(\text{def})}_{b,j},\,\mathbf{c}_h\bigr)
       + m_b
     \Bigr),
     \label{eq:triplet-safe} \\[2pt]
  \mathcal{L}_\text{unsafe}
  &= \frac{1}{n_h}\sum_{i=1}^{n_h}
     \operatorname{ReLU}\!\Bigl(
       d\!\bigl(\mathbf{h}^{(\text{def})}_{h,i},\,\mathbf{c}_h\bigr)
       - d\!\bigl(\mathbf{h}^{(\text{def})}_{h,i},\,\mathbf{h}^{(\text{base})}_{h,i}\bigr)
       + m_h
     \Bigr),
     \label{eq:triplet-unsafe}
\end{align}
each in standard triplet form
\(\operatorname{ReLU}\!\bigl(d(a,p) - d(a,n) + m\bigr)\):

\smallskip
\noindent
\begin{tabular}{@{}l l l l@{}}
                                       & Anchor                          & Positive (close)                  & Negative (far)                    \\
  Safe (Eq.~\ref{eq:triplet-safe})     & \(\mathbf{h}^{(\text{def})}_b\) & \(\mathbf{h}^{(\text{base})}_b\)  & \(\mathbf{c}_h\)                  \\
  Unsafe (Eq.~\ref{eq:triplet-unsafe}) & \(\mathbf{h}^{(\text{def})}_h\) & \(\mathbf{c}_h\)                  & \(\mathbf{h}^{(\text{base})}_h\)  \\
\end{tabular}

\smallskip
\noindent
The safe hinge keeps benign defended representations close to their
base counterpart while pushing them away from the harmful centroid
by margin \(m_b\!=\!500\).  The unsafe hinge clusters harmful
representations around the centroid while pushing them away from
their base counterpart by the larger margin \(m_h\!=\!1500\) (breaking
requires more displacement than anchoring).  The clipped-cosine term
in Eq.~\ref{eq:triplet-d} activates only when the angle exceeds
\(90^\circ\) and fades once representations diverge in direction.
